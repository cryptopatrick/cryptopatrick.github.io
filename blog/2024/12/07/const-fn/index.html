In Rust 1.63 (which dropped in August 2022, we got a nice addition; Mutex::new() and RwLock::new() became const fn. So what does this mean?

use std::sync::Mutex;

#[derive(Debug)]
struct Log {
    timestamp: &'static str,
    message: String,
}

// Vec::new() and String::new() do _not_ allocate so they do not go on the Heap.
// This means that we can work with them on the compiled side of the program.
// Mutex means that we can change the value inside, making the _static_ mutable.
static BATTLELOG: Mutex<Vec<Log>> = Mutex::new(Vec::new());

fn log_message(timestamp: &'static str) {
    // BATTLELOG is a global 
    BATTLELOG.lock().unwrap().push(Log{
        timestamp,
        message: "Some message".to_string(),
    });
}

fn main() {
    log_message("2024-12-07::09.00");
    log_message("2024-12-07::11.00");
    println!("{BATTLELOG:#?}");
}
