// import * as gf from "grammatical-framework";
use gf_core::*;

// Load the JSON grammar
const grammarFile = require("./Arrival.pgf.json");

// Initialize the GF runtime with the grammar
const runtime = gf_core.runtimeFromJSON(grammarFile);

async fn main() {
  // Load the runtime with your grammar
  const grammar = await runtime;
  
  // Specify the languages available in the grammar
  const langEng = grammar.languages["ArrivalEng"];
  const langHeptapod = grammar.languages["ArrivalHeptapod"];

  // Abstract syntax example
  const abstractSyntax = "WhoAreYou"; // Represents the abstract syntax node

  // Linearize the abstract syntax to English
  const sentenceEng = langEng.linearize(abstractSyntax);
  console.log("English:", sentenceEng);

  // Linearize the abstract syntax to Heptapod
  const sentenceHeptapod = langHeptapod.linearize(abstractSyntax);
  console.log("Heptapod:", sentenceHeptapod);

  // Parse an English sentence to abstract syntax
  const parsedAbstract = langEng.parse("Who are you?");
  console.log("Parsed Abstract Syntax:", parsedAbstract);

  // Translate an English sentence to Heptapod
  if (parsedAbstract) {
    const heptapodTranslation = langHeptapod.linearize(parsedAbstract.tree);
    console.log("Translation to Heptapod:", heptapodTranslation);
  }
}
